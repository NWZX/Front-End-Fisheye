/**
 * Function that check if email is valid
 * @param {string} email
 */
export const isEmail = (email) => {
    // Global RegExp for email validation
    const regexEmail = /^(([^<>()\\[\]\\.,;:\s@"]+(\.[^<>()\\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
    return regexEmail.test(email);
}

/**
 * Check if string is a integer
 * @param {string} value
 */
export const isInteger = (value) => {
    return !isNaN(value) &&
        parseInt(Number(value)) == value &&
        !isNaN(parseInt(value, 10));
}

/**
 * Trap the tab sequence inside a element
 * @param {KeyboardEvent} event 
 * @param {NodeListOf<Element>} focusableContent 
 */
export const captureTab = (event, focusableContent) => {
    const firstFocusableElement = focusableContent[0];
    const lastFocusableElement = focusableContent[focusableContent.length - 1];
  
    if (event.key === "Tab") {
        if (event.shiftKey) { // if shift key pressed for shift + tab combination
            if (document.activeElement === firstFocusableElement) {
                lastFocusableElement.focus(); // add focus for the last focusable element
                event.preventDefault();
            }
        } else { // if tab key is pressed
            if (document.activeElement === lastFocusableElement) { // if focused has reached to last focusable element then focus first focusable element after pressing tab
                firstFocusableElement.focus(); // add focus for the first focusable element
                event.preventDefault();
            }
        }
    }
}
